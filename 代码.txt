#include<stdio.h>             /*任意三个数中找出最大值*/
#include<stdlib.h>

int main()
{
    int a,b,c,max;             
    printf("请输入三个整数:");
    scanf("%d%d%d",&a, &b, &c);
    if(a>b)
        max=a;
    else max=b;
    if(max<c)
        max=c;
    printf("最大整数:max=%d\n",max);
    return 0;
}


#include<stdio.h>          /*求1000以内自然数的3的倍数的和*/
#include<stdlib.h>

int main()
{
    int sum=0, i=3; 
    while(i<=1000){
        sum+=i;
        i=i+3;
    }
    printf("3+6+9+...+999=%d\n",sum);
    return 0;
}



#include<stdio.h>            /*计算100以内整数之和*/
#include<stdlib.h>

int main()
{
    int sum=0, i=1;   
    while(i<=100){
        sum+=i;
        i++;
    }
    printf("1+2+3+...+100=%d\n",sum);
    return 0;
}


#include<stdio.h>    /*输入商品数量与单价，计算商品总额*/
int main()
{
    int amount;
    float price, total;
    printf("请输入数量:");
    scanf("%d",&amount);
    printf("请输入单价:");
    scanf("%f",&price);
    total=amount*price;
    printf("数量:%d,单价:%f,总额:%f\n",amount,price,total);
    return 0;
}


#include<stdio.h>   /*根据学生成绩做出五段评价*/
#include<stdlib.h>

int main()
{
    float score;
    printf("请输入成绩:");
    scanf("%f",&score);
    if (score>=90)
        printf("优秀.\n");
    else if (score>=80)
        printf("良好.\n");
    else if (score>=70)
        printf("中等.\n");
    else if (score>=60)
        printf("及格.\n");
    else
        printf("不及格.\n");
    return 0;
}
